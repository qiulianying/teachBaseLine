箭头函数

首先先来撸清楚箭头函数和普通函数写法上的区别 

//无参数
let fn = function(){
    return 'helloWorld';
}
//简写为：
let fn = ()=>{//但是没有参数时，括号不可以省略
    return 'helloWorld';
}
//根据规则二，简写为：
let fn = ()=>'helloWorld';



//一个参数
let fn = function(a){
    return a;
}
//简写为：
let fn = (a)=>{
    return a;
}
//根据规则一，还可以简写为：
let fn = a=>{
    return a;
}
//根据规则二，还可以简写为：
let fn = a=>a;



//多个参数
let fn = function(a,b){
    return a+b;
}
//简写为：
let fn = (a,b)=>{//多于一个参数，圆括号不可省略
    return a+b;
}
//根据规则二，还可以简写为：
let fn = (a,b)=>a+b;
